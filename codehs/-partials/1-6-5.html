1.6.5 Night Out

import java.util.Scanner; public class NightOut { public static void main(String[] args) { Scanner input = new Scanner(System.in); System.out.println("How much did dinner cost? "); double dinner = input.nextDouble(); System.out.println("How much is mini-golf for one person? "); double miniGolf = input.nextDouble(); System.out.println("How much did dessert cost? "); double dessert = input.nextDouble(); System.out.println("Dinner: $" + dinner); System.out.println("Mini-Golf: $" + miniGolf); System.out.println("Dessert: $" + dessert); System.out.println("Grand Total: $" + (dinner + miniGolf*2 + dessert)); } }
