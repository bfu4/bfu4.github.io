3.1.6 Number Order

import java.util.Scanner; public class RelativeNumbers { public static void main(String[] args) { System.out.println("Enter two numbers:"); Scanner input = new Scanner(System.in); int in1 = input.nextInt(); int in2 = input.nextInt(); // Compare the numbers as instructed System.out.println(in1 + " < " + in2 + ": " + (in1 < in2)); System.out.println(in1 + " == " + in2 + ": " + (in1 == in2)); System.out.println(in1 + " > " + in2 + ": " + (in1 > in2)); // Display the results } }
