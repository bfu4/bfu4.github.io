2.10.7 The Unit Circle

public class UnitCircle { public static void main(String[] args) { System.out.println("Radians: (cos, sin)"); //multiply by 100, round, and then divide by 100 to get 2 decimal points System.out.println("0.0: " + (Math.round(Math.cos(0.0)*100)/100) + ", " + (Math.round(Math.sin(0.0)*100)/100)); System.out.println("1.5707963267948966: " + (Math.round(Math.cos(1.5707963267948966)*100)/100) + ", " + (Math.round(Math.sin(1.5707963267948966)*100)/100)); System.out.println("3.141592653589793: " + (Math.round(Math.cos(3.141592653589793)*100)/100) + ", " + (Math.round(Math.sin(3.141592653589793)*100)/100)); } }
