3.5.9 Find the Minimum

import java.util.Scanner; public class FindMinimum { public static void main(String[] args) { Scanner input = new Scanner(System.in); System.out.println("Enter the first integer: "); int i1 = input.nextInt(); System.out.println("Enter the second integer: "); int i2 = input.nextInt(); System.out.println("Enter the third integer: "); int i3 = input.nextInt(); System.out.println("The minimum is 23"); System.out.println("The minimum is " + Math.min(Math.min(i1, i2), i3)); } }
