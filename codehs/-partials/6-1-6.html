6.1.6 Our First Array

public class FirstArray { public static void main(String[] args) { String[] cities = new String[] { "Las Vegas", "Minsk", "Sao Paulo" }; int[] temps = new int[] { 104, 73, 80 }; double[] precipitation = new double[] { 4.17, 26.7, 55.0 }; for(int i = 0; i < cities.length; i++) { System.out.println(cities[i] + " has an average annual precipitation of " + precipitation[i] + " inches."); System.out.println(cities[i] + " has an average annual high temp of " + temps[i] + " degrees Fahrenheit."); } } }
